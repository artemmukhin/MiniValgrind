{
    int len = 8;
    arr A[8];
    A[0] = 22; A[1] = 5; A[2] = 6;
    A[3] = 3; A[4] = 10; A[5] = 7;
    A[6] = 15; A[7] = 1;

    int i = 0;
    while (i < len - 1) {
        int j = 0;
        while (j < len - i - 1) {
            if (A[j] > A[j+1]) {
                int temp = A[j];
                A[j] = A[j+1];
                A[j+1] = temp;
            }
            j = j + 1;
        }
        i = i + 1;
    }
    print(A);
}
